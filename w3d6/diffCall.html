function Rabbit(name) {
this.name = name;
}
Rabbit.prototype.sayHi = function() {
alert(this.name);
};

let rabbit = new Rabbit("Rabbit");
 These calls do the same thing or not?

rabbit.sayHi(); //Answer: Rabbit
Rabbit.prototype.sayHi(); //Answer: undefined
Object.getPrototypeOf(rabbit).sayHi(); //Answer: undefined
rabbit.__proto__.sayHi(); //Answer: undefined
